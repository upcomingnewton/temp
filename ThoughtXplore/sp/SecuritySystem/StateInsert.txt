CREATE OR REPLACE FUNCTION Security_StateInsert(
StateName VARCHAR,
StateDescription VARCHAR,
by_user INTEGER,
ip VARCHAR,
OUT result INTEGER,
OUT rescode INTEGER
) RETURNS RECORD AS $$
DECLARE 
    temp INTEGER;
BEGIN 
temp := 0;

IF EXISTS ( SELECT id FROM "txMisc_miscstate" WHERE "StateName"=StateName) THEN
	result := 2;
  	rescode := 22;
	RETURN;
END IF;

INSERT INTO "txMisc_miscstate"( "StateName", "StateDescription")
    						VALUES (StateName,StateDescription);

GET DIAGNOSTICS temp := ROW_COUNT;
IF temp <> 1 THEN 
	rescode := 23;
	result := -1;
        RETURN;
ELSE
	SELECT id INTO rescode FROM "txMisc_miscstate" WHERE "StateName" = StateName;
	result := 1;
END IF;
RETURN;
END;
$$ LANGUAGE plpgsql;
